    let records = JSON.parse(localStorage.getItem('yuRecords')) || [];

    function updateView() {
        const historyList = document.getElementById('historyList');
        const summaryAmount = document.getElementById('summaryAmount');
        const summaryLabel = document.getElementById('summaryLabel');
        const btnSettleAll = document.getElementById('btnSettleAll');
        
        historyList.innerHTML = '<h3 style="font-size:1rem; color:var(--yu-heart);">足跡（履歴）</h3>';
        let totalYuPaid = 0, totalOtherPaid = 0, unpaidCount = 0;

        records.slice().reverse().forEach((rec, originalIndex) => {
            const index = records.length - 1 - originalIndex;
            const item = document.createElement('div');
            item.className = 'history-item';
            const payerName = rec.payer === 'yu' ? '優ちゃん' : 'あなた';
            const noteText = rec.note ? rec.note : '（内容なし）';
            
            item.innerHTML = `
                <div class="history-info">
                    <div class="history-note">${noteText}</div>
                    <strong style="color:${rec.payer === 'yu' ? '#ff69b4' : '#333'}">${rec.amount.toLocaleString()}円</strong> 
                    <span style="font-size:0.7rem; color:#666;">by ${payerName}</span><br>
                    <small style="color:#999;">${rec.date}</small>
                </div>
                ${rec.paid ? '<div class="paid-badge">完了</div>' : `<div class="unpaid-badge" onclick="togglePaid(${index})">精算</div>`}
            `;
            historyList.appendChild(item);

            if (!rec.paid) {
                unpaidCount++;
                if (rec.payer === 'yu') totalYuPaid += Number(rec.amount);
                else totalOtherPaid += Number(rec.amount);
            }
        });

        const diff = (totalYuPaid - totalOtherPaid) / 2;
        if (diff > 0) {
            summaryLabel.innerText = "あなたが優ちゃんに支払う額";
            summaryAmount.innerText = `${Math.abs(diff).toLocaleString()}円`;
        } else if (diff < 0) {
            summaryLabel.innerText = "優ちゃんからの「ありがとう」の重み";
            summaryAmount.innerText = `${Math.abs(diff).toLocaleString()}円分`;
        } else {
            summaryLabel.innerText = "清算の必要はありません";
            summaryAmount.innerText = "0円";
        }

        btnSettleAll.style.display = unpaidCount > 0 ? 'block' : 'none';
        localStorage.setItem('yuRecords', JSON.stringify(records));
    }

    function addRecord() {
        const noteEl = document.getElementById('note');
        const amountEl = document.getElementById('amount');
        const payerEl = document.getElementById('payer');
        if (!amountEl.value || amountEl.value <= 0) return;

        records.push({
            note: noteEl.value,
            amount: parseInt(amountEl.value),
            payer: payerEl.value,
            date: new Date().toLocaleString('ja-JP', {month:'short', day:'numeric', hour:'2-digit', minute:'2-digit'}),
            paid: false
        });
        
        noteEl.value = '';
        amountEl.value = '';
        document.getElementById('yuSpeech').innerText = `「${yuQuotes[Math.floor(Math.random() * yuQuotes.length)]}」`;
        document.getElementById('yuImage').src = yuImages[Math.floor(Math.random() * yuImages.length)];
        updateView();
    }

    function togglePaid(index) {
        records[index].paid = true;
        updateView();
    }

    function settleAll() {
        if(confirm('未精算の項目をすべて清算済みにしますか？')) {
            records.forEach(rec => rec.paid = true);
            updateView();
            document.getElementById('yuSpeech').innerText = "「清算完了！次も楽しみだね？」";
        }
    }

    function clearAll() {
        if(confirm('全ての記録をリセットして、新しいページをめくりますか？')) {
            records = [];
            updateView();
            document.getElementById('yuSpeech').innerText = "「また新しい思い出、作ろうね？」";
        }
    }

    updateView();
</script>
